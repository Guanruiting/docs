merge: function(first, second) {
 var l = second.length,
     i = first.length,
     j = 0;

 if (typeof l === "number") {
   for (; j < l; j++) {
     first[i++] = second[j];
   }
 } else {
   while (second[j] !== undefined) {
     first[i++] = second[j++];
   }
 }

 first.length = i;
